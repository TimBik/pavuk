    function onFormDemoRequestSubmit() {
      var allInputs = jQuery(".popup-content :input");
      allInputs.each(function() {
        jQuery(this).val(jQuery.trim(jQuery(this).val()));
      });
      var email = document.getElementById("drEmail");
      var phone = document.getElementById("drPhone");
      var name = document.getElementById("drName");

      jQuery('#drEmail').val(function () {
        return this.value.toLowerCase();
      });
      email.classList.remove('fill-required');
      phone.classList.remove('fill-required');
      name.classList.remove('fill-required');

      if (!email.checkValidity()) {
        email.classList.add('fill-required');
        email.placeholder = 'Необходимо указать email';
      }

      if (!phone.checkValidity()) {
        phone.classList.add('fill-required');
        phone.placeholder = 'Необходимо указать телефон';
      }

      if (!name.checkValidity()) {
        name.classList.add('fill-required');
        name.placeholder = 'Необходимо указать имя';
      }

      if (email.checkValidity() && phone.checkValidity() && name.checkValidity()) {
        grecaptcha.execute();
      }

      return false;
    }

      template = (isRail ? 'rail' : 'auto') + ':' + template;
